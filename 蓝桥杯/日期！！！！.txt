	public static void main(String[] args) throws IOException
	{
		LocalDate date =LocalDate.of(2239,9,9);
		LocalDate end =LocalDate.of(9876,1,1);
		long sum=0;
		while(date.compareTo(end)<0)
		{
			StringBuilder raw=new StringBuilder();
			
			raw.append(date.getYear());
			raw.append(date.getMonthValue());
			raw.append(date.getDayOfMonth());
			int[] times=new int[10];
			for(int i=0;i<raw.length();i++)times[raw.charAt(i)-'0']++;
			boolean flag=true;
			for(int i=0;i<=9;i++)
			{
				if(!flag)break;
				if(times[i]==0)continue;
				for(int j=0;j<=9;j++)
				{
					if(times[i]!=times[j]&&times[j]!=0)
					{
						flag=false;
						break;
					}
				}
			}
			if(flag)sum++;
			date=date.plusDays(1);
		}
		System.out.println(sum);
	}
		
}
