{	
	//Kruscal 模板
		
	public static class edge
	{
		int from;int to; int dis;
		edge(int from,int to,int dis)
		{
			this.from=from;
			this.to=to;
			this.dis=dis;
		}
	}
	
	public static ArrayList<edge> edges;
		
	public static int[] parents;
	
	public static int findRoot(int index)
	{
		if(parents[index]==index)return index;
		parents[index]=findRoot(parents[index]);
		return parents[index];	
	}
	
	public static void union(int rootA,int rootB)
	{
		parents[rootB]=rootA;
	}
	
	public static int Kruscal(int n)
	{
		int count=0;
		int sum=0;
		for(edge nowEdge : edges)
		{
			if(count>=n-1)break;
			int rootFrom=findRoot(nowEdge.from);
			int rootTo=findRoot(nowEdge.to);
			if(rootFrom!=rootTo)
			{
				sum+=nowEdge.dis;
				count++;
				union(rootFrom, rootTo);
			}
		}
		
		if(count<n-1)return -1;
		return sum;
	}
	
	
	public static void main(String[] args) throws IOException 
	{
		BufferedReader reader=new BufferedReader(new InputStreamReader(System.in));
		StringTokenizer tokenizer=new StringTokenizer(reader.readLine());
		int N=Integer.parseInt(tokenizer.nextToken());
		int M=Integer.parseInt(tokenizer.nextToken());
		edges=new ArrayList<>();
		parents=new int[N+1];
		for(int i=1;i<=N;i++)parents[i]=i;
		for(int i=0;i<M;i++)
		{
			StringTokenizer t=new StringTokenizer(reader.readLine());
			int from=Integer.parseInt(t.nextToken());
			int to=Integer.parseInt(t.nextToken());
			int dis=Integer.parseInt(t.nextToken());
			if(from!=to)
			{
				edges.add(new edge(from, to, dis));
				edges.add(new edge(to, from, dis));
			}
		}
		edges.sort((a,b)->Integer.compare(a.dis,b.dis));
		int ans=Kruscal(N);
		if(ans==-1)System.out.println("orz");
		else System.out.println(ans);
	}
	
	
	
	
	
	
		
}
