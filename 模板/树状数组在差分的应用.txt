	//差分+树状数组深入理解
	public static long[] TreeMap;
	public static int lowbit(int x) {return x&-x;}
	public static void add(int x,long k,int n)//单点更新方便初始化
	{
		int i1=x;
		while(i1<=n)
		{
			TreeMap[i1]+=k;
			i1+=lowbit(i1);//lowbit表示当前层数 加一个当前层数就表示下一层
		}
	}
	public static long ask(int x)
	{
		long sum=0;
		int i=x;
		while(i>0)
		{
			sum+=TreeMap[i];
			i-=lowbit(i);
		}
		return sum;
	}
