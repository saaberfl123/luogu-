public class Main 
{
	public static class edge
	{
		int to;
		long dis;
		public edge(int to,long dis) {
			this.to=to;this.dis=dis;
		}
	}
	public static void main(String[] args) throws IOException 
	{
		BufferedReader reader=new BufferedReader(new InputStreamReader(System.in));
		PrintWriter writer=new PrintWriter(new OutputStreamWriter(System.out));
		StringTokenizer tokenizer=new StringTokenizer(reader.readLine());
		int n=Integer.parseInt(tokenizer.nextToken());
		int m=Integer.parseInt(tokenizer.nextToken());
		int s=Integer.parseInt(tokenizer.nextToken());
		long[] dist=new long[n+1];
		boolean[] isVisited=new boolean[n+1];
		Arrays.fill(dist,Long.MAX_VALUE);
		dist[s]=0;
		ArrayList<edge>[] map=new ArrayList[n+1];
		for(int i=1;i<=n;i++)map[i]=new ArrayList<>();
		for(int i=0;i<m;i++)
		{
			StringTokenizer t=new StringTokenizer(reader.readLine());
			int from=Integer.parseInt(t.nextToken());
			int to=Integer.parseInt(t.nextToken());
			int dis=Integer.parseInt(t.nextToken());
			if(from!=to)//环
			{
				map[from].add(new edge(to,dis));
			}
			
		}
		Queue<long[]> queue=new PriorityQueue<>((a,b)->Long.compare(a[0], b[0]));
		queue.offer(new long[]{0,s});//dis to
		while(!queue.isEmpty())
		{
			long[] a=queue.poll();//每次都找到最小值
			long dis=a[0];
			int index=(int)a[1];
			if(isVisited[index])continue;
			isVisited[index]=true;
			for(edge nowEdge:map[index])
			{
				if(dist[index]+nowEdge.dis<dist[nowEdge.to])
				{
					dist[nowEdge.to]=dist[index]+nowEdge.dis;
					queue.offer(new long[]{dist[nowEdge.to],nowEdge.to});
				}
			}
		}
		for(int i=1;i<=n;i++)writer.print(dist[i]+" ");
		writer.flush();
		
	}

}
