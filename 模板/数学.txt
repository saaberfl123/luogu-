public static long fast(int a,int b,int p)
{
	long res=1;long base=a;//幂的话就不要用int了很容易溢出
	while(b>0)
	{
		if((b&1)==1)//最低位是1res=(res*base)%p;
		base=(base*base)%p;//这里运用了模的性质 如是全乘 则模运算可以分配到每一项中
		b=b>>1;//去掉最后一位
		return res%p;
	}
}
