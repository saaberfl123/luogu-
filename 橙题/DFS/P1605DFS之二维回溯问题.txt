public class Main 
{
	public static int count=0;
	public static int[][]  map;
	public static int[] dx= {1,-1,0,0};
	public static int[] dy= {0,0,1,-1};//方向数组
	public static void dfs(int x,int y,int n,int m)
	{
		if(x>n||y>m||x<=0||y<=0||map[x][y]==1)return;
		if(map[x][y]==2)
		{
			count++;
			return;
		}
		for(int i=0;i<4;i++)
		{
			map[x][y]=1;
			dfs(x+dx[i], y+dy[i], n, m);
			map[x][y]=0;
		}
	}
	public static void main(String[] args) 
	{
		Scanner scanner=new Scanner(System.in);
		int n=scanner.nextInt();
		int m=scanner.nextInt();
		int t=scanner.nextInt();
		int SX=scanner.nextInt();int SY=scanner.nextInt();
		int FX=scanner.nextInt();int FY=scanner.nextInt();
		map=new int[n+1][m+1];map[FX][FY]=2;
		for(int i=0;i<t;i++)
		{
			int x=scanner.nextInt();
			int y=scanner.nextInt();
			map[x][y]=1;
		}
		dfs(SX, SY, n, m);
		System.out.println(count);
	}
}
